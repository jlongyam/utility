/**
 * @namespace env 
 * @returns {Object} 
 *
 */
let env = (function () {
  const isBrowser =
    typeof window === "object" &amp;&amp; typeof document !== "undefined";
  const isNode =
    typeof process !== "undefined" &amp;&amp;
    process.release &amp;&amp;
    process.release.name === "node";
  const isWorker =
    typeof self === "object" &amp;&amp;
    self.constructor &amp;&amp;
    self.constructor.name &amp;&amp;
    self.constructor.name.endsWith("WorkerGlobalScope");

  return {
    browser: isBrowser,
    node: isNode,
    worker: isWorker,
  };
})();

env.global = (typeof window !== "undefined") ? window
  : (typeof global !== "undefined") ? global
    : (typeof self !== "undefined") ? self
      : (typeof globalThis !== "undefined") ? globalThis
        : this


export default env;
